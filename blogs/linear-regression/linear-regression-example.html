numerator = sum((x_train - x_mean)*(y_train - y_mean))
denominator = sum((x_train - x_mean)**2)
beta_1 = numerator / denominator
beta_0 = y_mean - beta_1*x_mean

y_pred = beta_0 + beta_1*x_train
MSE = (sum((y_pred - y_train)**2))/len(y_train)

y_pred_test = beta_0 + beta_1*x_test
MSE_test = (sum((y_pred_test - y_test)**2))/len(y_test)

print('The parameters are:\n beta_0 = {} \n beta_1 = {} \n train MSE: {} \n test MSE: {}'.format(round(beta_0,4), round(beta_1,4), round(MSE,4), round(MSE_test,4)))

# linear regression with 1,000 iterations

b_0 = 0.0
b_1 = 0.0
n = len(x_train)
alpha = 0.0001
for i in range(1000):
    y_pred = b_0 + b_1*x_train
    error = y_pred - y_train
    mse = sum(error**2)/n
    b_0 -= (2/n) * alpha*sum(error)
    b_1 -= (2/n) * alpha*sum(error*x_train)
    
y_pred_gd1 = b_0 + b_1*x_train
MSE_gd1 = (sum((y_pred_gd1 - y_train)**2))/len(y_train)

y_pred_gd1_test = b_0 + b_1*x_test
MSE_gd1_test = (sum((y_pred_gd1_test - y_test)**2))/len(y_test)
    
print('The parameters are:\n b_0 = {} \n b_1 = {} \n train MSE: {} \n train MSE: {}'.format(round(b_0,4), round(b_1,4), round(MSE_gd1,4), round(MSE_gd1_test,4)))
